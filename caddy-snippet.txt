# BirdBoard — Extrait de configuration Caddy
# A ajouter dans votre Caddyfile existant, dans le bloc de votre site.
# Remplacez {USER} par votre nom d'utilisateur systeme.

handle /birds/api/* {
    # Proxy vers le backend Node.js
    uri strip_prefix /birds
    reverse_proxy 127.0.0.1:7474
}

handle /birds/audio/* {
    uri strip_prefix /birds/audio
    root * /home/{USER}/BirdSongs/Extracted
    file_server
}

handle /birds* {
    root * /home/{USER}/pibird
    file_server
    try_files {path} /birds/index.html
}
